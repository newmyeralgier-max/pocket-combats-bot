"""
victory_combo_v4.py
Белая тема. Логика 1–12 строго по чеклисту Moon.
Ключевые моменты:
- adb_back используется ТОЛЬКО при активном оверлее (syschat/monster_params).
- Вкладки управляются по имени и состоянию chevron справа от названия (open_tab/close_tab).
- Если Attack не найден: сначала проверяем start.png; если якорь есть — листаем вниз и повторяем нормализацию вкладок.
- Бой: moves → preffered_skill / preffered_skill_text → default.
- Победа: ждем Victory и жмем Continue; если Continue не найден — НЕ жмем Back.
- Лут: по items_header. Клики по именам — относительные позиции; кнопка "подобрать" — HSV с мягким решением в пользу нажатия.
- Никаких “img2 = screencap() or img” — только безопасная подстановка.

Пути и имена шаблонов не менять:
- Скрипт:   C:\bot\script\victory_combo_v4.py
- Шаблоны:  C:\bot\tpl\my\  (имена строго как в твоей папке)
- Логи:     C:\bot\log\victory_YYYY-MM-DD.txt
- Скрины:   C:\bot\log\snapshots\
"""

# ====== ADB ======
ADB_SERIAL = os.getenv("ADB_SERIAL", "").strip() or None

def _adb(args: List[str], timeout=8):
    base = ["adb"]
    if ADB_SERIAL:
        base += ["-s", ADB_SERIAL]
    return subprocess.run(base + args, stdout=subprocess.PIPE, stderr=subprocess.PIPE, timeout=timeout)

def adb_tap(x: int, y: int, delay=0.35):
    _adb(["shell", "input", "tap", str(int(x)), str(int(y))])
    time.sleep(delay)

def adb_swipe(x1: int, y1: int, x2: int, y2: int, duration_ms=280, delay=0.6):
    _adb(["shell", "input", "swipe", str(int(x1)), str(int(y1)), str(int(x2)), str(int(y2)), str(int(duration_ms))])
    time.sleep(delay)

def adb_back(delay=0.35):
    _adb(["shell", "input", "keyevent", "4"])
    time.sleep(delay)

def screencap() -> Optional[np.ndarray]:
    try:
        proc = _adb(["exec-out", "screencap", "-p"], timeout=6)
        data = np.frombuffer(proc.stdout, np.uint8)
        img = cv2.imdecode(data, cv2.IMREAD_COLOR)
        if img is not None:
            return img
    except Exception:
        pass
    try:
        remote = "/sdcard/__scr_v4.png"
        _adb(["shell", "screencap", "-p", remote], timeout=6)
        local = os.path.join(SNAP_DIR, "__last.png")
        _adb(["pull", remote, local], timeout=6)
        img = cv2.imread(local, cv2.IMREAD_COLOR)
        return img
    except Exception:
        return None

def save_snap(img: np.ndarray, tag: str):
    path = os.path.join(SNAP_DIR, f"{int(time.time())}_{tag}.png")
    try:
        cv2.imwrite(path, img)
    except Exception:
        pass

# ====== Шаблоны (как в твоей папке) ======
class TEMPLS:
    START               = "start.png"
    SYSCHAT_TEXT        = "syschat.png"
    MONSTER_PARAMS_TEXT = "monster_params.png"

    CLOSE1 = "close_1.png"
    CLOSE2 = "close_2.png"

    MONSTER_TAB  = "monster_tab.png"
    ITEMS_TAB    = "items_tab.png"
    ITEMS_HEADER = "items_header.png"

    TAB_OPEN  = "open_tab.png"     # стрелка/иконка: вкладка раскрыта
    TAB_CLOSE = "close_tab.png"    # стрелка/иконка: вкладка свернута

    ATTACK_ICON      = "attack.png"
    MOVES            = "moves.png"
    PREF_SKILL       = "preffered_skill.png"
    PREF_SKILL_TEXT  = "preffered_skill_text.png"
    DEFAULT          = "default.png"

    VICTORY  = "victory.png"
    CONTINUE = "continue.png"

THRESH_DEFAULT = 0.85
THRESH_STRONG  = 0.89  # тексты/шапки

def load_templ(name: str) -> Optional[np.ndarray]:
    p = os.path.join(TPL_DIR, name)
    if not os.path.exists(p):
        log.warning(f"Нет шаблона: {name}")
        return None
    return cv2.imread(p, cv2.IMREAD_COLOR)

def match_full(screen: np.ndarray, name: str, thr: float) -> Optional[Tuple[int,int,int,int,float]]:
    tpl = load_templ(name)
    if tpl is None or screen is None:
        return None
    res = cv2.matchTemplate(screen, tpl, cv2.TM_CCOEFF_NORMED)
    _, max_val, _, max_loc = cv2.minMaxLoc(res)
    if max_val >= thr:
        h, w = tpl.shape[:2]
        x, y = max_loc
        return (x, y, w, h, float(max_val))
    return None

def match_in_roi(screen: np.ndarray, name: str, thr: float, roi: Tuple[int,int,int,int]) -> Optional[Tuple[int,int,int,int,float]]:
    tpl = load_templ(name)
    if tpl is None or screen is None:
        return None
    x1, y1, x2, y2 = roi
    x1 = max(0, min(x1, screen.shape[1]-1))
    x2 = max(0, min(x2, screen.shape[1]))
    y1 = max(0, min(y1, screen.shape[0]-1))
    y2 = max(0, min(y2, screen.shape[0]))
    if x2 <= x1 or y2 <= y1: return None
    crop = screen[y1:y2, x1:x2]
    res = cv2.matchTemplate(crop, tpl, cv2.TM_CCOEFF_NORMED)
    _, max_val, _, max_loc = cv2.minMaxLoc(res)
    if max_val >= thr:
        h, w = tpl.shape[:2]
        rx, ry = max_loc
        return (x1 + rx, y1 + ry, w, h, float(max_val))
    return None

def center(rect: Tuple[int,int,int,int]) -> Tuple[int,int]:
    x, y, w, h = rect
    return (x + w//2, y + h//2)

# ====== Свайпы ======
SWIPE_DOWN = (540, 1000, 540, 400, 280)
SWIPE_UP   = (540, 400, 540, 1000, 280)

def swipe_down(): adb_swipe(*SWIPE_DOWN)
def swipe_up():   adb_swipe(*SWIPE_UP)

# ====== Якоря состояний ======
def on_start_screen(img: np.ndarray) -> bool:
    return match_full(img, TEMPLS.START, THRESH_STRONG) is not None

def is_overlay_active(img: np.ndarray) -> bool:
    return (match_full(img, TEMPLS.SYSCHAT_TEXT, THRESH_STRONG) is not None) or \
           (match_full(img, TEMPLS.MONSTER_PARAMS_TEXT, THRESH_STRONG) is not None)

# ====== Вкладки: состояние по chevron справа от текста вкладки ======
def detect_tab_bbox(img: np.ndarray, tab_tpl: str, thr=THRESH_DEFAULT) -> Optional[Tuple[int,int,int,int]]:
    m = match_full(img, tab_tpl, thr)
    if not m: return None
    x, y, w, h, _ = m
    return (x, y, w, h)

def detect_tab_state(img: np.ndarray, tab_tpl: str) -> Optional[str]:
    """
    Возвращает 'open' / 'close' / None. Ищем chevron только в узкой полосе справа от текста вкладки.
    """
    box = detect_tab_bbox(img, tab_tpl)
    if not box: return None
    x, y, w, h = box
    H, W = img.shape[:2]
    x1 = min(x + w + 2, W-1)
    x2 = min(x + w + int(0.10 * W), W)
    y1, y2 = y, min(y + h, H)
    if x2 <= x1 or y2 <= y1: return None
    roi = (x1, y1, x2, y2)

    open_m  = match_in_roi(img, TEMPLS.TAB_OPEN,  THRESH_DEFAULT, roi)
    close_m = match_in_roi(img, TEMPLS.TAB_CLOSE, THRESH_DEFAULT, roi)

    if open_m and (not close_m or open_m[4] >= close_m[4]): return "open"
    if close_m: return "close"
    return None

def scroll_until_visible(tab_tpl: str, thr=THRESH_DEFAULT, max_up=5, max_down=0) -> bool:
    img = screencap()
    if img is None: return False
    if detect_tab_bbox(img, tab_tpl, thr): return True
    for _ in range(max_up):
        swipe_up()
        img = screencap()
        if img is None: continue
        if detect_tab_bbox(img, tab_tpl, thr): return True
    for _ in range(max_down):
        swipe_down()
        img = screencap()
        if img is None: continue
        if detect_tab_bbox(img, tab_tpl, thr): return True
    return False

def ensure_tab_open(tab_tpl: str):
    if not scroll_until_visible(tab_tpl): return
    img = screencap()
    if img is None: return
    state = detect_tab_state(img, tab_tpl)
    if tab_tpl == TEMPLS.ITEMS_TAB and match_full(img, TEMPLS.ITEMS_HEADER, THRESH_STRONG):
        state = "open"
    if state == "open":
        log.info(f"Вкладка {tab_tpl} уже открыта")
        return
    log.info(f"Вкладка {tab_tpl} → открываю по имени")
    box = detect_tab_bbox(img, tab_tpl)
    if box: adb_tap(*center(box))

def ensure_tab_close(tab_tpl: str):
    if not scroll_until_visible(tab_tpl): return
    img = screencap()
    if img is None: return
    state = detect_tab_state(img, tab_tpl)
    if state == "close":
        log.info(f"Вкладка {tab_tpl} уже закрыта")
        return
    log.info(f"Вкладка {tab_tpl} → закрываю по имени")
    box = detect_tab_bbox(img, tab_tpl)
    if box: adb_tap(*center(box))

# ====== Логика "монстры открыты?" / "вещи открыты?" ======
def any_attack_visible(img: np.ndarray) -> bool:
    return match_full(img, TEMPLS.ATTACK_ICON, THRESH_DEFAULT) is not None

def is_items_open(img: np.ndarray) -> bool:
    return match_full(img, TEMPLS.ITEMS_HEADER, THRESH_STRONG) is not None

def ensure_items_closed():
    img = screencap()
    if img is None: return
    if is_items_open(img):
        ensure_tab_close(TEMPLS.ITEMS_TAB)

def ensure_monsters_open():
    img = screencap()
    if img is None: return
    if not any_attack_visible(img):
        ensure_tab_open(TEMPLS.MONSTER_TAB)

def close_monsters():
    img = screencap()
    if img is None: return
    if any_attack_visible(img):
        ensure_tab_close(TEMPLS.MONSTER_TAB)

def open_items():
    img = screencap()
    if img is None: return
    if not is_items_open(img):
        ensure_tab_open(TEMPLS.ITEMS_TAB)

# ====== HSV проверка 'ПОДОБРАТЬ' (мягкая в пользу активации) ======
PICKUP_REL_ROI = (0.62, 0.70, 0.92, 0.86)
HSV_BROWN = ((5, 90, 40), (25, 255, 255))    # активная
HSV_GRAY  = ((0, 0, 70), (180, 30, 200))     # неактивная

def abs_roi(img: np.ndarray, rel_roi):
    h, w = img.shape[:2]
    x1, y1, x2, y2 = rel_roi
    return (int(w*x1), int(h*y1), int(w*x2), int(h*y2))

def ratio_hsv(img_bgr, low, high):
    hsv = cv2.cvtColor(img_bgr, cv2.COLOR_BGR2HSV)
    mask = cv2.inRange(hsv, np.array(low, dtype=np.uint8), np.array(high, dtype=np.uint8))
    return float(np.count_nonzero(mask)) / float(mask.size)

def pickup_is_active(img: np.ndarray) -> bool:
    x1, y1, x2, y2 = abs_roi(img, PICKUP_REL_ROI)
    roi = img[y1:y2, x1:x2]
    if roi.size == 0: return True  # fail-open: лучше попробовать нажать
    brown = ratio_hsv(roi, *HSV_BROWN)
    gray  = ratio_hsv(roi, *HSV_GRAY)
    # Мягкое правило: если серого мало (<0.02) — считаем активной; иначе активна, если коричневого больше серого на 0.015
    if gray < 0.02:
        return True
    return (brown - gray) > 0.015

# ====== Вспомогательные ======
def wait_for_battle(max_wait=8.0, poll=0.3) -> bool:
    t_end = time.time() + max_wait
    while time.time() < t_end:
        img = screencap()
        if img is None:
            time.sleep(poll); continue
        if match_full(img, TEMPLS.MOVES, THRESH_DEFAULT) or match_full(img, TEMPLS.DEFAULT, THRESH_DEFAULT):
            return True
        time.sleep(poll)
    return False

# ====== Шаги 1–12 ======
def step1_close_overlays():
    img = screencap()
    if img is None: return
    # syschat
    if match_full(img, TEMPLS.SYSCHAT_TEXT, THRESH_STRONG):
        log.info("Syschat обнаружен → закрываю (только тут можно Back)")
        c1 = match_full(img, TEMPLS.CLOSE1, THRESH_DEFAULT)
        c2 = match_full(img, TEMPLS.CLOSE2, THRESH_DEFAULT)
        if c1: adb_tap(*center(c1[:4]))
        elif c2: adb_tap(*center(c2[:4]))
        else: adb_back()
        time.sleep(0.5)

    img = screencap()
    if img is None: return
    # monster params
    if match_full(img, TEMPLS.MONSTER_PARAMS_TEXT, THRESH_STRONG):
        log.info("Monster params обнаружен → закрываю (только тут можно Back)")
        c1 = match_full(img, TEMPLS.CLOSE1, THRESH_DEFAULT)
        c2 = match_full(img, TEMPLS.CLOSE2, THRESH_DEFAULT)
        if c1: adb_tap(*center(c1[:4]))
        elif c2: adb_tap(*center(c2[:4]))
        else: adb_back()
        time.sleep(0.5)

def step2_normalize_tabs():
    log.info("Нормализую вкладки: Вещи закрыть, Монстры открыть")
    # Возвращаем вкладки наверх экрана
    scroll_until_visible(TEMPLS.MONSTER_TAB)
    scroll_until_visible(TEMPLS.ITEMS_TAB)
    # Строго по состоянию chevron
    ensure_tab_close(TEMPLS.ITEMS_TAB)
    ensure_tab_open(TEMPLS.MONSTER_TAB)

def step3_scroll_down():
    swipe_down()
    log.info("Листнул вниз")

def step4_tap_attack_icon():
    img = screencap()
    if img is None:
        log.info("Нет скрина → фоллбек Attack")
        adb_tap(920, 560); return

    # Если не видим Attack, сперва проверим start.png
    pos = match_full(img, TEMPLS.ATTACK_ICON, THRESH_DEFAULT)
    if not pos:
        if on_start_screen(img):
            log.info("На старте: листаю вниз и проверяю вкладки заново")
            swipe_down()
            step2_normalize_tabs()
            img2 = screencap()
            if img2 is None:
                adb_tap(920, 560); return
            pos = match_full(img2, TEMPLS.ATTACK_ICON, THRESH_DEFAULT)
            if pos:
                log.info("Жму Attack (после нормализации)")
                adb_tap(*center(pos[:4]))
            else:
                log.info("Attack не найден → фоллбек")
                adb_tap(920, 560)
        else:
            # Возможно боевой экран/оверлей — не жмём Back! просто пробуем фоллбек Attack
            log.info("Не старт, Attack не виден → фоллбек Attack")
            adb_tap(920, 560)
    else:
        log.info("Жму Attack (по шаблону)")
        adb_tap(*center(pos[:4]))

    # Ждём вход в бой
    if not wait_for_battle():
        log.info("После Attack бой не начался вовремя")

def step5_battle_moves_skill_default():
    t_end = time.time() + 10.0
    while time.time() < t_end:
        img = screencap()
        if img is None:
            time.sleep(0.2); continue
        m = match_full(img, TEMPLS.MOVES, THRESH_DEFAULT)
        if m:
            adb_tap(*center(m[:4])); time.sleep(0.5)
            img2 = screencap()
            if img2 is None:
                img2 = img
            s = match_full(img2, TEMPLS.PREF_SKILL, THRESH_DEFAULT) or match_full(img2, TEMPLS.PREF_SKILL_TEXT, THRESH_DEFAULT)
            if s:
                log.info("Бой: Preferred skill")
                adb_tap(*center(s[:4])); return
            d = match_full(img2, TEMPLS.DEFAULT, THRESH_DEFAULT)
            if d:
                log.info("Бой: Default")
                adb_tap(*center(d[:4])); return
            time.sleep(0.5); continue
        else:
            d = match_full(img, TEMPLS.DEFAULT, THRESH_DEFAULT)
            if d:
                log.info("Бой: Default (moves не найден)")
                adb_tap(*center(d[:4])); return
            time.sleep(0.4)
    log.info("Бой: не удалось нажать skill/attack вовремя")

def step6_victory_and_continue():
    log.info("Жду победу...")
    t_end = time.time() + 25.0
    while time.time() < t_end:
        img = screencap()
        if img is None: time.sleep(0.2); continue
        if match_full(img, TEMPLS.VICTORY, THRESH_DEFAULT):
            log.info("Победа!"); break
        time.sleep(0.4)

    log.info("Ищу Continue...")
    t2 = time.time() + 8.0
    found = False
    while time.time() < t2:
        img = screencap()
        if img is None: time.sleep(0.2); continue
        c = match_full(img, TEMPLS.CONTINUE, THRESH_DEFAULT)
        if c:
            adb_tap(*center(c[:4])); time.sleep(0.8)
            found = True
            break
        time.sleep(0.3)
    if not found:
        log.info("Continue не найден — пропускаю Back, чтобы не выйти из игры")

def step7_scroll_down(): step3_scroll_down()

def step8_close_monsters():
    log.info("Закрываю Монстры (по состоянию chevron)")
    ensure_tab_close(TEMPLS.MONSTER_TAB)

def step9_open_items():
    log.info("Открываю Вещи (по состоянию chevron)")
    ensure_tab_open(TEMPLS.ITEMS_TAB)

# ====== Лут: имена предметов от items_header ======
def iter_item_name_positions(img: np.ndarray) -> List[Tuple[int,int]]:
    h, w = img.shape[:2]
    hdr = match_full(img, TEMPLS.ITEMS_HEADER, THRESH_STRONG)
    if not hdr:
        return []
    x, y, ww, hh, _ = hdr
    base_x = x + int(0.06 * w)
    base_y = y + hh + int(0.02 * h)
    row_h  = int(0.085 * h)  # при необходимости подстрой 0.08–0.095
    return [(base_x, base_y + i*row_h) for i in range(5)]

def step10_loot_first_then_rule():
    open_items()
    img0 = screencap()
    if img0 is None: return
    if not match_full(img0, TEMPLS.ITEMS_HEADER, THRESH_STRONG):
        scroll_until_visible(TEMPLS.ITEMS_HEADER, THRESH_STRONG, max_up=5, max_down=0)
        img0 = screencap() or img0
    positions = iter_item_name_positions(img0)
    if not positions:
        log.info("Не вижу items_header → лут пропущен"); return

    for idx, (x, y) in enumerate(positions, start=1):
        log.info(f"Лут #{idx}: раскрываю предмет (тап по имени)")
        adb_tap(x, y, delay=0.6)
        img = screencap()
        if img is None:
            time.sleep(0.3); continue
        if pickup_is_active(img):
            x1, y1, x2, y2 = abs_roi(img, PICKUP_REL_ROI)
            cx, cy = (x1 + x2)//2, (y1 + y2)//2
            log.info("Лут: 'ПОДОБРАТЬ' активна → жму")
            adb_tap(cx, cy, delay=0.7)
            time.sleep(0.4)
            continue
        else:
            log.info("Лут: 'ПОДОБРАТЬ' неактивна → сворачиваю и выхожу из лутинга")
            adb_tap(x, y, delay=0.5)
            break

def step11_restore_initial_view():
    log.info("Восстанавливаю исходный вид: Вещи закрыть → Монстры открыть → свайп вниз")
    ensure_tab_close(TEMPLS.ITEMS_TAB)
    ensure_tab_open(TEMPLS.MONSTER_TAB)
    step3_scroll_down()

# ====== Главный цикл ======
def run_cycle():
    # 1–11 по чеклисту
    step1_close_overlays()
    step2_normalize_tabs()
    step3_scroll_down()
    step4_tap_attack_icon()
    step5_battle_moves_skill_default()
    step6_victory_and_continue()
    step7_scroll_down()
    step8_close_monsters()
    step9_open_items()
    step10_loot_first_then_rule()
    step11_restore_initial_view()

def main():
    log.info("Старт victory_combo_v4 — цикл боя и лутинга")
    i = 0
    while True:
        i += 1
        log.info(f"=== Цикл #{i} ===")
        try:
            run_cycle()
        except KeyboardInterrupt:
            log.info("Остановлено пользователем (Ctrl+C)")
            break
        except Exception as e:
            log.exception(f"Ошибка цикла: {e}")
            time.sleep(1.0)
        time.sleep(0.7)

if __name__ == "__main__":
    main()
