import json
import re
from pathlib import Path
from typing import List, Tuple, Dict, Set

# === Пути к конфигу ===
CFG_DIR = Path(__file__).resolve().parent.parent / "cfg"
CONFIG_PATH = CFG_DIR / "config.json"

# === Служебные суффиксы ===
SUFFIXES = ("_icon", "_name", "_full")

# === Кириллица и базовые утилиты ===
CYR_RE = re.compile(r"[А-Яа-яЁё]")

def has_cyrillic(s: str) -> bool:
    return bool(CYR_RE.search(s))

def base_cyr_from_filename(fname: str) -> str:
    # Берём имя файла без пути, убираем индекс в конце и служебные суффиксы, сохраняем кириллицу
    stem = Path(fname).stem
    stem = re.sub(r"_(\d+)$", "", stem)
    for suf in SUFFIXES:
        if stem.endswith(suf):
            stem = stem[: -len(suf)]
            break
    # Нормализуем мягко (оставляем кириллицу), приводим к нижнему, заменяем пробелы
    s = stem.lower().replace("ё", "е").replace(" ", "_")
    return re.sub(r"[^a-z0-9а-я_]", "", s)

# === Загрузка/сохранение ===
def load_json(path: Path):
    with open(path, "r", encoding="utf-8") as f:
        return json.load(f)

def save_config(cfg: dict):
    # Страховка по ключам
    if "ALLOWED_ITEM_NAMES" not in cfg or not isinstance(cfg["ALLOWED_ITEM_NAMES"], list):
        cfg["ALLOWED_ITEM_NAMES"] = []
    if "SKIP_ITEM_NAMES" not in cfg or not isinstance(cfg["SKIP_ITEM_NAMES"], list):
        cfg["SKIP_ITEM_NAMES"] = []

    # Дедуп и сортировка
    cfg["ALLOWED_ITEM_NAMES"] = sorted(set(cfg["ALLOWED_ITEM_NAMES"]))
    cfg["SKIP_ITEM_NAMES"] = sorted(set(cfg["SKIP_ITEM_NAMES"]))

    with open(CONFIG_PATH, "w", encoding="utf-8") as f:
        json.dump(cfg, f, ensure_ascii=False, indent=2)

# === Сканирование PNG (только 4 папки) ===
def list_png_in_dirs(dirs: List[Path]) -> List[str]:
    files: List[str] = []
    for d in dirs:
        if not d.exists():
            continue
        for p in d.rglob("*.png"):
            files.append(p.name)
    return sorted(set(files))

# === Пагинация выбора групп ===
def choose_groups_paged(group_items: List[Tuple[str, List[str]]], page_size: int = 10) -> Set[int]:
    selected: Set[int] = set()
    total = len(group_items)
    idx = 0
    while idx < total:
        page_start = idx
        page_end = min(idx + page_size, total)
        print(f"\nГруппы (стр. {page_start // page_size + 1}):")
        for i in range(page_start, page_end):
            base, variants = group_items[i]
            print(f"{i+1}. {base} ({len(variants)} файла)")
            for v in variants:
                print(f"    - {v}")
        sel = input("Номера через запятую (Enter — далее, a — все, q — закончить): ").strip().lower()
        if sel == "q":
            break
        if sel == "a":
            selected.update(range(page_start, total))
            break
        if sel:
            for part in sel.split(","):
                part = part.strip()
                if part.isdigit():
                    gi = int(part) - 1
                    if page_start <= gi < page_end:
                        selected.add(gi)
        idx = page_end
    return selected

# === Режим ввода ===
def mode_prompt() -> str:
    while True:
        mode = input("Выберите режим (add/remove/sync): ").strip().lower()
        if mode in ("add", "remove", "sync"):
            return mode

# === Основная логика ===
def main():
    # Жёстко фиксируем 4 папки-источника
    tpl_dirs = [
        Path(r"C:\bot\tpl\иконки_рун"),
        Path(r"C:\bot\tpl\иконки_предметов"),
        Path(r"C:\bot\tpl\имя_предметов"),
        Path(r"C:\bot\tpl\описание_рун"),
    ]

    print("[DEBUG] директории источника:")
    for d in tpl_dirs:
        print("   ", d, "— существует:", d.exists())

    if not CONFIG_PATH.exists():
        print("config.json не найден")
        return

    cfg = load_json(CONFIG_PATH)
    # Страховка ключей
    if "ALLOWED_ITEM_NAMES" not in cfg: cfg["ALLOWED_ITEM_NAMES"] = []
    if "SKIP_ITEM_NAMES" not in cfg: cfg["SKIP_ITEM_NAMES"] = []

    # Скан файлов только в 4 папках
    files = list_png_in_dirs(tpl_dirs)
    print(f"[DEBUG] найдено файлов: {len(files)}")

    # Формируем группы по русскому базовому имени и отбрасываем всё без кириллицы
    groups_map: Dict[str, List[str]] = {}
    for fname in files:
        base = base_cyr_from_filename(fname)
        if not has_cyrillic(base):
            continue
        groups_map.setdefault(base, []).append(fname)

    # Упорядочим группы и вариации
    group_items = sorted(((k, sorted(v)) for k, v in groups_map.items()), key=lambda x: x[0])
    print(f"[DEBUG] групп (рус.): {len(group_items)}")

    mode = mode_prompt()

    if mode == "add":
        if not group_items:
            print("Нет групп для добавления.")
            print("Готово.")
            return

        selected = choose_groups_paged(group_items, page_size=10)

        added_files = 0
        for gi in sorted(selected):
            _, variants = group_items[gi]
            for v in variants:
                if v not in cfg["ALLOWED_ITEM_NAMES"]:
                    cfg["ALLOWED_ITEM_NAMES"].append(v)
                    added_files += 1
                if v in cfg["SKIP_ITEM_NAMES"]:
                    cfg["SKIP_ITEM_NAMES"].remove(v)

        save_config(cfg)
        print(f"[ИТОГ] Добавлено групп: {len(selected)}, файлов: {added_files}")
        print("Готово.")

    elif mode == "remove":
        allowed = list(cfg.get("ALLOWED_ITEM_NAMES", []))
        if not allowed:
            print("ALLOWED пуст. Удалять нечего.")
            print("Готово.")
            return

        remove_groups: Dict[str, List[str]] = {}
        for v in allowed:
            base = base_cyr_from_filename(v)
            if not has_cyrillic(base):
                continue
            remove_groups.setdefault(base, []).append(v)

        remove_items = sorted(((k, sorted(v)) for k, v in remove_groups.items()), key=lambda x: x[0])
        if not remove_items:
            print("Нет русских групп в ALLOWED для удаления.")
            print("Готово.")
            return

        selected = choose_groups_paged(remove_items, page_size=10)

        removed_files = 0
        if selected:
            keep = []
            remove_set = set()
            for gi in sorted(selected):
                _, variants = remove_items[gi]
                remove_set.update(variants)
            for v in cfg["ALLOWED_ITEM_NAMES"]:
                if v not in remove_set:
                    keep.append(v)
                else:
                    removed_files += 1
            cfg["ALLOWED_ITEM_NAMES"] = keep

        save_config(cfg)
        print(f"[ИТОГ] Удалено групп: {len(selected)}, файлов: {removed_files}")
        print("Готово.")

    elif mode == "sync":
        allowed = []
        for _, variants in group_items:
            allowed.extend(variants)
        cfg["ALLOWED_ITEM_NAMES"] = sorted(set(allowed))
        cfg["SKIP_ITEM_NAMES"] = []

        save_config(cfg)
        print(f"[ИТОГ] Синхронизация: ALLOWED={len(cfg['ALLOWED_ITEM_NAMES'])}, SKIP=0")
        print("Готово.")

if __name__ == "__main__":
    main()
