# victory_combo.py — универсальный бот: Wi‑Fi + USB, шаблоны, логи, опциональный OCR
# Минимум зависимостей: cv2/numpy (для шаблонов), pytesseract — опционально

import os
import sys
import time
import logging
import subprocess

# ---- Конфигурация ----
DEVICE_IP = '192.168.0.100:5555'  # '' если не хочешь Wi‑Fi-подключение
USE_WIFI_FIRST = True
ATTACK_TEMPLATE = 'attack.png'    # твой файл шаблона
THRESHOLD = 0.82                  # порог совпадения шаблона (0..1)
LOOP_DELAY = 1.5                  # пауза между циклами (сек)
DRY_RUN = False                    # True = НЕ нажимаем; False = нажимаем
DEBUG_DIR = 'debug'
LOG_DIR = 'logs'
SCREEN_PATH = os.path.join(DEBUG_DIR, 'last.png')
LOG_PATH = os.path.join(LOG_DIR, 'run.log')

# ---- Логирование ----
os.makedirs(DEBUG_DIR, exist_ok=True)
os.makedirs(LOG_DIR, exist_ok=True)
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s | %(levelname)s | %(message)s',
    handlers=[logging.FileHandler(LOG_PATH, encoding='utf-8'), logging.StreamHandler(sys.stdout)]
)
log = logging.getLogger("victory_combo")

# ---- Опциональные зависимости ----
HAVE_CV2 = True
try:
    import cv2
    import numpy as np
except Exception as e:
    HAVE_CV2 = False
    log.warning("cv2/numpy недоступны — поиск шаблонов отключён. Установи: pip install opencv-python-headless numpy")

HAVE_TESS = True
try:
    import pytesseract  # Для Windows tesseract.exe нужен отдельно; иначе пропустим
except Exception:
    HAVE_TESS = False
    log.info("pytesseract недоступен — OCR отключён (не критично).")

# ---- Вспомогательные функции ADB ----
def adb_available() -> bool:
    try:
        subprocess.run(['adb', 'version'], stdout=subprocess.PIPE, stderr=subprocess.PIPE, timeout=5)
        return True
    except Exception as e:
        log.error("ADB недоступен. Установи ADB и добавь в PATH.")
        return False

def adb_run(args, serial=None, timeout=15, capture=False):
    cmd = ['adb']
    if serial:
        cmd += ['-s', serial]
    cmd += args
    try:
        if capture:
            res = subprocess.run(cmd, stdout=subprocess.PIPE, stderr=subprocess.PIPE, timeout=timeout)
            return res.returncode, res.stdout, res.stderr
        else:
            res = subprocess.run(cmd, stdout=subprocess.PIPE, stderr=subprocess.PIPE, timeout=timeout)
            return res.returncode, res.stdout.decode('utf-8', errors='ignore'), res.stderr.decode('utf-8', errors='ignore')
    except subprocess.TimeoutExpired:
        return 124, b'' if capture else '', 'timeout'
    except Exception as e:
        return 1, b'' if capture else '', str(e)

def list_devices():
    code, out, err = adb_run(['devices'])
    if code != 0:
        log.error(f"adb devices error: {err}")
        return []
    devs = []
    for line in out.splitlines():
        line = line.strip()
        if '\tdevice' in line and not line.startswith('List of'):
            devs.append(line.split('\t')[0])
    return devs

def connect_wifi(ip_port: str):
    log.info(f"[wifi] подключение к {ip_port}...")
    code, out, err = adb_run(['connect', ip_port])
    if code == 0 and 'connected to' in out or 'already connected' in out.lower():
        log.info(out.strip())
        return True
    log.warning(f"Не удалось подключиться по Wi‑Fi: {out or err}")
    return False

def ensure_device(ip_pref: str, use_wifi_first=True):
    # 1) Попробовать Wi‑Fi
    if use_wifi_first and ip_pref:
        if connect_wifi(ip_pref):
            # Проверим, что видим устройство в списке
            devs = list_devices()
            if ip_pref in devs:
                return ip_pref
    # 2) Иначе — взять первое доступное устройство
    devs = list_devices()
    if devs:
        if ip_pref and ip_pref in devs:
            return ip_pref
        return devs[0]
    return None

def capture_screen(serial, out_path):
    os.makedirs(os.path.dirname(out_path), exist_ok=True)
    log.info(f"[screen] снимаю экран → {out_path}")
    try:
        with open(out_path, 'wb') as f:
            res = subprocess.run(['adb', '-s', serial, 'exec-out', 'screencap', '-p'], stdout=f, stderr=subprocess.PIPE, timeout=20)
        if res.returncode != 0:
            log.error(f"screencap error: {res.stderr.decode('utf-8', errors='ignore')}")
            return False
        return True
    except Exception as e:
        log.error(f"screencap exception: {e}")
        return False

def tap(serial, x, y):
    if DRY_RUN:
        log.info(f"[tap][DRY] ({x},{y}) — нажатие не выполнено")
        return True
    code, out, err = adb_run(['shell', 'input', 'tap', str(int(x)), str(int(y))], serial=serial)
    if code == 0:
        log.info(f"[tap] ({x},{y}) — OK")
        return True
    log.error(f"[tap] ошибка: {err or out}")
    return False

# ---- Визуальные функции ----
def find_template(screen_path, template_path, threshold=0.82):
    if not HAVE_CV2:
        return None
    if not os.path.exists(template_path):
        log.warning(f"Нет файла шаблона: {template_path}")
        return None
    screen = cv2.imread(screen_path, cv2.IMREAD_COLOR)
    templ = cv2.imread(template_path, cv2.IMREAD_COLOR)
    if screen is None or templ is None:
        log.error("Не удалось прочитать screen или шаблон (PNG/JPG?)")
        return None
    res = cv2.matchTemplate(screen, templ, cv2.TM_CCOEFF_NORMED)
    min_val, max_val, min_loc, max_loc = cv2.minMaxLoc(res)
    h, w = templ.shape[:2]
    center = (max_loc[0] + w / 2, max_loc[1] + h / 2)
    log.info(f"[templ] max={max_val:.3f} at {max_loc}, size={w}x{h}")
    if max_val >= threshold:
        return center, max_val
    return None

def ocr_debug(screen_path):
    if not HAVE_TESS:
        return ""
    try:
        import cv2  # в случае, если cv2 есть, можно серым сделать
        img = cv2.imread(screen_path)
        if img is None:
            return ""
        text = pytesseract.image_to_string(img)
        text = text.strip()
        if text:
            clean_text = text[:120].replace('\n', ' / ')
            log.info(f"Торг {clean_text}...")

        return text
    except Exception as e:
        log.debug(f"OCR skip: {e}")
        return ""

# ---- Основной цикл ----
def main():
    if not adb_available():
        return

    serial = ensure_device(DEVICE_IP, USE_WIFI_FIRST)
    if not serial:
        log.error("Нет доступных устройств. Подключи USB или укажи правильный IP:PORT.")
        return
    log.info(f"[device] работаю с: {serial}")

    # Проверка шаблона
    if HAVE_CV2 and not os.path.exists(ATTACK_TEMPLATE):
        log.warning(f"Внимание: {ATTACK_TEMPLATE} не найден. Положи рядом с скриптом.")

    # Цикл работы
    try:
        cycle = 0
        while True:
            cycle += 1
            log.info(f"--- Цикл #{cycle} ---")

            ok = capture_screen(serial, SCREEN_PATH)
            if not ok:
                log.warning("Пробую переподключиться...")
                serial = ensure_device(DEVICE_IP, USE_WIFI_FIRST)
                time.sleep(1.0)
                continue

            # OCR для диагностики (не обязателен)
            ocr_debug(SCREEN_PATH)

            # Поиск кнопки атаки
            hit = find_template(SCREEN_PATH, ATTACK_TEMPLATE, THRESHOLD) if HAVE_CV2 else None
            if hit:
                (cx, cy), score = hit
                log.info(f"[attack] цель найдена ({cx:.1f},{cy:.1f}) score={score:.3f}")
                tap(serial, cx, cy)
            else:
                log.info("[attack] шаблон не найден — пропуск")

            time.sleep(LOOP_DELAY)
    except KeyboardInterrupt:
        log.info("Остановлено пользователем (Ctrl+C).")
    except Exception as e:
        log.exception(f"Критическая ошибка: {e}")

if __name__ == "__main__":
    main()
