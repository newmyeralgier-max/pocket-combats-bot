# adb_utils.py — чистая версия
import os
import re
import time
import subprocess
from typing import Optional

# ---------- низкоуровневые вызовы ADB ----------

def _adb_cmd(serial: str, *args: str):
    return ["adb", "-s", serial, *args]

from typing import List
def _adb_out(serial: str, args: List[str], decode: bool = True):

    """Возвращает stdout. Если decode=False — сырые bytes."""
    proc = subprocess.run(_adb_cmd(serial, *args), stdout=subprocess.PIPE, stderr=subprocess.STDOUT)
    if decode:
        return proc.stdout.decode("utf-8", errors="ignore")
    return proc.stdout

# ---------- базовые утилиты ----------

def tap(serial: str, x: int, y: int):
    _adb_out(serial, ["shell", "input", "tap", str(x), str(y)])

def swipe(serial: str, x1: int, y1: int, x2: int, y2: int, duration_ms: int = 200):
    _adb_out(serial, ["shell", "input", "swipe", str(x1), str(y1), str(x2), str(y2), str(duration_ms)])

def screenshot(serial: str, save_to: Optional[str] = None):
    """
    Делает PNG-скрин с экрана. Если save_to указан — сохраняет файл и возвращает путь.
    Если нет — возвращает bytes PNG.
    """
    png = _adb_out(serial, ["exec-out", "screencap", "-p"], decode=False)
    if save_to:
        os.makedirs(os.path.dirname(save_to), exist_ok=True)
        with open(save_to, "wb") as f:
            f.write(png)
        return save_to
    return png

# ---------- пакет/активность (foreground) ----------

def _extract_pkg(text: str) -> str:
    """
    Пытается вытащить имя пакета из разных строк dumpsys/cmd.
    Собираем первый удачный матч вида 'package/Class'.
    """
    patterns = [
        r"mCurrentFocus=.*?\s([A-Za-z0-9\._]+)/",            # dumpsys window windows
        r"mFocusedApp.*?ActivityRecord\{.*?\s([A-Za-z0-9\._]+)/",
        r"topResumedActivity:.*?\s([A-Za-z0-9\._]+)/",       # dumpsys activity activities
        r"ResumedActivity:.*?\s([A-Za-z0-9\._]+)/",
        r"mResumedActivity:.*?\s([A-Za-z0-9\._]+)/",
        r"\bACTIVITY\s+([A-Za-z0-9\._]+)/",                  # cmd activity top
    ]
    for pat in patterns:
        m = re.search(pat, text)
        if m:
            return m.group(1)
    return ""

def get_current_package(serial: str) -> str:
    # 1) Самый частый источник
    try:
        out = _adb_out(serial, ["shell", "dumpsys", "window", "windows"])
        pkg = _extract_pkg(out)
        if pkg:
            return pkg
    except Exception:
        pass
    # 2) Альтернативный источник для Android 10+
    try:
        out = _adb_out(serial, ["shell", "dumpsys", "activity", "activities"])
        pkg = _extract_pkg(out)
        if pkg:
            return pkg
    except Exception:
        pass
    # 3) Еще один вариант
    try:
        out = _adb_out(serial, ["shell", "cmd", "activity", "top"])
        pkg = _extract_pkg(out)
        if pkg:
            return pkg
    except Exception:
        pass

    return ""

def is_process_running(serial: str, package: str) -> bool:
    # Сначала pidof (быстро и просто)
    try:
        out = _adb_out(serial, ["shell", "pidof", package])
        if out.strip():
            return True
    except Exception:
        pass
    # Запасной метод — ps | grep
    try:
        out = _adb_out(serial, ["shell", "ps", "-A"])
        for line in out.splitlines():
            # точное вхождение пакета (хитрый grep не нужен)
            if package in line and "grep" not in line:
                return True
    except Exception:
        pass
    return False

def launch_game(serial: str, package: str, activity: Optional[str] = None):
    """
    Запускает игру.
    - Если известно activity (например, '.MainActivity'), стартуем напрямую.
    - Иначе используем monkey с LAUNCHER activity.
    """
    try:
        if activity:
            # Если активити начинается с точки — допишем пакет
            comp = f"{package}/{activity}" if activity.startswith(".") else f"{package}/{activity}"
            _adb_out(serial, ["shell", "am", "start", "-n", comp])
        else:
            _adb_out(serial, ["shell", "monkey", "-p", package, "-c", "android.intent.category.LAUNCHER", "1"])
    except Exception as e:
        print(f"[launch_game] error: {e}")
