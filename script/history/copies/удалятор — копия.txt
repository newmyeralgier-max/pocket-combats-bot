import os

# ===== НАСТРОЙКИ =====
FOLDER_PATH = r"c:\bot\tpl\названия"  # Папка, из которой удаляем
NAME_PART = "rej_"                       # Подстрока в имени файла для удаления
FILE_FORMAT = "png"                        # "all" — любой формат, или "png"/"jpg"/"txt" и т.д.

# ===== ЛОГИКА =====
def delete_files(folder, name_part, file_format):
    if not os.path.isdir(folder):
        print(f"[ERR] Папка не найдена: {folder}")
        return

    deleted = 0
    for fname in os.listdir(folder):
        full_path = os.path.join(folder, fname)

        if not os.path.isfile(full_path):
            continue

        # Проверка подстроки в имени
        if name_part.lower() not in fname.lower():
            continue

        # Проверка формата
        if file_format.lower() != "all":
            if not fname.lower().endswith("." + file_format.lower()):
                continue

        try:
            os.remove(full_path)
            print(f"[DEL] {fname}")
            deleted += 1
        except Exception as e:
            print(f"[ERR] Не удалось удалить {fname}: {e}")

    print(f"[DONE] Удалено файлов: {deleted}")

# ===== ЗАПУСК =====
delete_files(FOLDER_PATH, NAME_PART, FILE_FORMAT)
