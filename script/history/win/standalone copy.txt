import cv2
import numpy as np
import unicodedata

# ===== ФЛАГИ ОТЛАДКИ =====
DEBUG_SAVE = True     # Сохранять промежуточные картинки
DEBUG_OCR  = True     # Печатать детальный OCR-лог

# ===== УТИЛИТЫ =====
def try_imread(path: str):
    """
    Чтение файла с поддержкой unicode-путей.
    Возвращает изображение BGR или None.
    """
    data = np.fromfile(path, dtype=np.uint8)
    return cv2.imdecode(data, cv2.IMREAD_COLOR)

def norm_name(name: str) -> str:
    """
    NFKC-нормализация, удаление управляющих символов, lowercase.
    """
    name = unicodedata.normalize("NFKC", name)
    name = "".join(c for c in name if unicodedata.category(c)[0] != "C")
    return name.strip().lower()

# ===== ОСНОВНАЯ ФУНКЦИЯ ПОИСКА =====
def find_loot_item_coords(img_bgr, safe_name, mon_name=None, ocr_data=None):
    """
    Ищет координаты предмета по нормализованному имени.
    ocr_data — словарь с ключами: text, left, top, width, height
    """
    # Проверки на None
    if ocr_data is None:
        raise ValueError("OCR data is None — передай результат OCR в функцию")

    required_keys = {"text", "left", "top", "width", "height"}
    missing_keys = required_keys - set(ocr_data)
    if missing_keys:
        raise KeyError(f"OCR data отсутствуют ключи: {missing_keys}")

    # Отладочный вывод
    if DEBUG_OCR:
        print(f"[OCR LOG] Монстр: {mon_name or '<не указан>'}")
        for word in ocr_data["text"]:
            if word.strip():
                print(f"{word} -> norm={norm_name(word)}")

    # 1. Точное совпадение
    for i, word in enumerate(ocr_data["text"]):
        if norm_name(word) == safe_name:
            x = ocr_data["left"][i]
            y = ocr_data["top"][i]
            w = ocr_data["width"][i]
            h = ocr_data["height"][i]
            if DEBUG_OCR:
                print(f"Точное совпадение: {word} @ ({x},{y},{w},{h})")
            return x, y, w, h

    # 2. (опционально) Частичное совпадение — можно добавить при необходимости
    # for i, word in enumerate(ocr_data["text"]):
    #     if safe_name in norm_name(word):
    #         ...

    if DEBUG_OCR:
        print(f"Совпадений для '{safe_name}' не найдено.")
    return None
