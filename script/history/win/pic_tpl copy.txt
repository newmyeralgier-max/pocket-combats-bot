import json
from pathlib import Path
import cv2
import numpy as np

MONSTERS_DIR = Path("tpl/монстры")
JSON_DIR     = Path("tpl/json")
OUT_DIR      = Path("tpl/debug_crops")
OUT_DIR.mkdir(parents=True, exist_ok=True)

def imread_unicode(path: Path):
    buf = np.fromfile(str(path), dtype=np.uint8)
    return cv2.imdecode(buf, cv2.IMREAD_COLOR)

for json_file in sorted(JSON_DIR.glob("*.json")):
    with open(json_file, encoding="utf-8") as f:
        data = json.load(f)

    coords = data.get("coords")
    if not coords or len(coords) != 4:
        print(f"[SKIP] {json_file.name} — нет coords")
        continue

    t, b, l, r = [int(v) for v in coords]
    img_path = Path(data["source"])
    if not img_path.is_absolute():
        img_path = MONSTERS_DIR / img_path.name

    img = imread_unicode(img_path)
    if img is None:
        print(f"[ERR] Не открыть {img_path}")
        continue

    roi = img[t:b, l:r]
    if roi.size == 0:
        print(f"[WARN] {json_file.stem} — пустой ROI {coords}")
        continue

    out_path = OUT_DIR / f"{json_file.stem}_ROI.png"
    cv2.imwrite(str(out_path), roi)
    print(f"[OK] {out_path.name} [{roi.shape[1]}x{roi.shape[0]}]")

print("\nГотово. Смотри кропы в", OUT_DIR)
